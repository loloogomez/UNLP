programa examen
procesos
  proceso avenidaoctaba (ES papeles: numero ;ES flores: numero)
  comenzar
    mientras HayPapelEnLaEsquina
      tomarPapel
      papeles:= papeles +1
    mientras HayFlorEnLaEsquina
      tomarFlor
      flores:= flores +1
    mover
  fin
  proceso rectangulo (ES flores: numero; ES papeles: numero)
  comenzar
    repetir 2
      repetir flores
        mover
      derecha
      repetir papeles
        mover
      derecha
  fin
areas
  ciudad: AreaC(1,1,100,100)
robots
  robot robot1
  variables
    papeles: numero
    flores: numero
  comenzar
    Pos(8,1)
    mientras (HayPapelEnLaEsquina | HayFlorEnLaEsquina)
      avenidaoctaba (papeles,flores)
    Pos(1,1)
    rectangulo  (flores, papeles)
  fin
variables
  R-info: robot1
comenzar
  AsignarArea(R-info,ciudad)
  Iniciar(R-info,1,1)
fin
