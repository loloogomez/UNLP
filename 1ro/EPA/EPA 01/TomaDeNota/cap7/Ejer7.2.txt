programa Ejer72
procesos
  proceso Cantidad (ES Flores:numero)
  variables
    FloresEsquina: numero
  comenzar 
    FloresEsquina:= 0
    mientras HayFlorEnLaEsquina
      tomarFlor
      Flores:= Flores +1
      FloresEsquina:= FloresEsquina +1
    repetir FloresEsquina
      depositarFlor
  fin
areas
  ciudad: AreaC(1,1,100,100)
robots
  robot robot1
  variables
    flores: numero
  comenzar
    repetir 7
      Cantidad (flores)
      repetir 99
        mover
        Cantidad (flores)
      Pos(PosAv +1,1)
    Informar(flores)
  fin
variables
  R-info: robot1
comenzar
  AsignarArea(R-info,ciudad)
  Iniciar(R-info,1,1)
fin
