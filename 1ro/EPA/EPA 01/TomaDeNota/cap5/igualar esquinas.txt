programa prueba
procesos
  proceso QueHayEnLaEsquina (ES flores: numero; ES papeles: numero)
  comenzar
    flores:= 0
    papeles:= 0
    mientras HayFlorEnLaEsquina
      tomarFlor
      flores:= flores +1
    repetir (flores)
      depositarFlor
    mientras HayPapelEnLaEsquina
      tomarPapel
      papeles:= papeles +1
    repetir (papeles)
      depositarPapel
  fin
  proceso Igualar (ES flores: numero; ES papeles: numero)
  comenzar
    mientras (flores < papeles)
      tomarPapel
      papeles:= papeles -1
    mientras (flores > papeles)
      tomarFlor
      flores:= flores -1
  fin
areas
  ciudad: AreaC(1,1,100,100)
robots
  robot robot1
  variables
    flores: numero
    papeles: numero
  comenzar
    repetir 99
      QueHayEnLaEsquina (flores,papeles)
      Igualar (flores,papeles)
      repetir 99
        mover
        QueHayEnLaEsquina (flores,papeles)
        Igualar (flores,papeles)
      Pos(PosAv +1, 1)   
  fin
variables
  R-info: robot1
comenzar
  AsignarArea(R-info,ciudad)
  Iniciar(R-info,1,1)
fin
